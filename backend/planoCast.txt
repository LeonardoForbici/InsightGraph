## 2. Sprint 1 — Call Resolver Real

### Por que isso é o mais importante?

Hoje seu parser detecta `import UserService` mas não sabe que `userService.findById(id)` CHAMA `UserService.findById`. O call resolver resolve isso — sem ele, todas as análises de impacto são incompletas.

### 2.1 Criar `backend/analyzers/call_resolver.py`

Crie o arquivo com este conteúdo COMPLETO:

```python
"""
call_resolver.py
Resolve chamadas de método reais a partir do AST do tree-sitter.
Transforma chamadas como userService.save(entity) em arestas CALLS reais no grafo.
"""

import re
from dataclasses import dataclass, field
from typing import Optional


@dataclass
class FieldDeclaration:
    """Representa um campo de classe: private UserService userService;"""
    field_name: str
    type_name: str
    class_owner: str


@dataclass
class MethodCallEdge:
    """Representa uma aresta CALLS resolvida."""
    from_key: str          # namespace_key do caller
    to_class: str          # nome da classe chamada
    to_method: str         # nome do método chamado
    resolved_key: Optional[str] = None  # namespace_key resolvido (se encontrar)
    confidence: str = "high"  # high / medium / low


class JavaCallResolver:
    """
    Resolve chamadas de método em Java usando o AST do tree-sitter.

    Estratégia:
    1. Primeira passagem: coleta todos os campos das classes (field declarations)
       ex: private UserService userService  →  {userService: UserService}
    2. Segunda passagem: para cada method_invocation, resolve o objeto receptor
       ex: userService.findById(id)  →  UserService.findById

    Limitações conhecidas (aceitáveis para análise local):
    - Não resolve chamadas encadeadas complexas (a.b().c())
    - Não resolve generics ou lambda
    - Herança não é rastreada (mas raramente importa para impacto)
    """

    def __init__(self):
        self.field_map: dict[str, dict[str, str]] = {}
        # {class_name: {field_name: type_name}}

    def collect_fields(self, class_node, class_name: str, content_bytes: bytes):
        """
        Primeira passagem: coleta todos os campos declarados na classe.
        Preenche self.field_map[class_name].
        """
        if class_name not in self.field_map:
            self.field_map[class_name] = {}

        def walk(node):
            if node.type == "field_declaration":
                type_node = node.child_by_field_name("type")
                declarator = node.child_by_field_name("declarator")
                if type_node and declarator:
                    type_name = content_bytes[type_node.start_byte:type_node.end_byte].decode("utf-8", errors="ignore")
                    name_node = declarator.child_by_field_name("name")
                    if name_node:
                        field_name = content_bytes[name_node.start_byte:name_node.end_byte].decode("utf-8", errors="ignore")
                        # Remove generics: List<User> → List
                        type_clean = re.sub(r"<.*>", "", type_name).strip()
                        self.field_map[class_name][field_name] = type_clean

            # Também captura parâmetros do construtor com @Autowired
            if node.type == "constructor_declaration":
                params_node = node.child_by_field_name("parameters")
                if params_node:
                    for param in params_node.children:
                        if param.type == "formal_parameter":
                            ptype = param.child_by_field_name("type")
                            pname = param.child_by_field_name("name")
                            if ptype and pname:
                                type_name = content_bytes[ptype.start_byte:ptype.end_byte].decode("utf-8", errors="ignore")
                                field_name = content_bytes[pname.start_byte:pname.end_byte].decode("utf-8", errors="ignore")
                                type_clean = re.sub(r"<.*>", "", type_name).strip()
                                self.field_map[class_name][field_name] = type_clean

            for child in node.children:
                walk(child)

        walk(class_node)

    def resolve_calls_in_method(
        self,
        method_node,
        class_name: str,
        method_name: str,
        method_ns_key: str,
        content_bytes: bytes
    ) -> list[MethodCallEdge]:
        """
        Segunda passagem: encontra todas as method_invocations no body do método
        e tenta resolver para qual classe/método elas apontam.
        """
        calls = []
        fields = self.field_map.get(class_name, {})

        def walk(node):
            if node.type == "method_invocation":
                obj_node = node.child_by_field_name("object")
                name_node = node.child_by_field_name("name")

                if obj_node and name_node:
                    obj_text = content_bytes[obj_node.start_byte:obj_node.end_byte].decode("utf-8", errors="ignore").strip()
                    called_method = content_bytes[name_node.start_byte:name_node.end_byte].decode("utf-8", errors="ignore").strip()

                    # Tenta resolver o tipo do objeto receptor
                    resolved_class = None

                    # Caso 1: campo da classe (this.userService ou userService)
                    obj_clean = obj_text.replace("this.", "")
                    if obj_clean in fields:
                        resolved_class = fields[obj_clean]

                    # Caso 2: variável local com nome padrão (ex: userRepo → UserRepository)
                    # Heurística: primeira letra maiúscula + "Repository", "Service", etc.
                    if not resolved_class:
                        for suffix in ["Repository", "Service", "Controller", "Manager", "Handler", "Mapper", "Dao"]:
                            if obj_clean.lower().endswith(suffix.lower()):
                                # Capitaliza a primeira letra
                                resolved_class = obj_clean[0].upper() + obj_clean[1:]
                                break

                    # Caso 3: nome da variável bate com algum campo em case-insensitive
                    if not resolved_class:
                        for fname, ftype in fields.items():
                            if fname.lower() == obj_clean.lower():
                                resolved_class = ftype
                                break

                    if resolved_class and called_method:
                        edge = MethodCallEdge(
                            from_key=method_ns_key,
                            to_class=resolved_class,
                            to_method=called_method,
                            confidence="high" if obj_clean in fields else "medium"
                        )
                        calls.append(edge)

            for child in node.children:
                walk(child)

        walk(method_node)
        return calls


def resolve_ts_calls(
    file_content: str,
    current_component_key: str,
    imported_components: dict[str, str]
) -> list[MethodCallEdge]:
    """
    Resolver simples para TypeScript/React.
    Detecta chamadas como: useRouter(), fetch('/api/...'), componentName()

    imported_components = {"useRouter": "next/router:useRouter", "UserCard": "components/UserCard:UserCard"}
    """
    calls = []

    # Padrão: identifier(  — captura chamadas de função
    call_pattern = re.compile(r'\b([A-Za-z_][A-Za-z0-9_]*)\s*\(')

    for match in call_pattern.finditer(file_content):
        func_name = match.group(1)
        if func_name in imported_components:
            calls.append(MethodCallEdge(
                from_key=current_component_key,
                to_class=func_name,
                to_method=func_name,
                resolved_key=imported_components[func_name],
                confidence="high"
            ))

    return calls
```

### 2.3 Adicionar `resolve_pending_calls` no `main.py`

Esta função roda APÓS todo o scan e resolve os `to_unresolved` contra os nós conhecidos:

```python
async def resolve_pending_calls(all_entities: dict) -> None:
    """
    Após o scan completo, resolve chamadas pendentes que têm `to_unresolved`.
    ex: {to_unresolved: "UserService.save"} → busca nó com name="save" e parent_class="UserService"
    """
    global memory_nodes

    # Índice: "ClassName.methodName" → namespace_key
    method_index: dict[str, str] = {}
    for node in memory_nodes:
        if "Java_Method" in node.get("labels", []):
            parent = node.get("parent_class", "")
            name = node.get("name", "")
            if parent and name:
                method_index[f"{parent}.{name}"] = node["namespace_key"]

    resolved_count = 0
    for rel in all_entities.get("relationships", []):
        if "to_unresolved" in rel:
            unresolved = rel.pop("to_unresolved")
            confidence = rel.pop("confidence", "medium")
            if unresolved in method_index:
                rel["to"] = method_index[unresolved]
                rel["props"] = {"confidence": confidence}
                resolved_count += 1
            else:
                # Não encontrou o nó destino — cria um nó fantasma (external)
                # para mostrar dependência com sistema externo
                ghost_key = f"external:{unresolved}"
                class_name, method_name = unresolved.rsplit(".", 1) if "." in unresolved else (unresolved, "unknown")
                memory_nodes.append({
                    "namespace_key": ghost_key,
                    "name": method_name,
                    "labels": ["Java_Method", "Entity", "External"],
                    "layer": "External",
                    "project": "external",
                    "parent_class": class_name,
                })
                rel["to"] = ghost_key

    logger.info("Call resolver: %d/%d chamadas resolvidas", resolved_count, len([r for r in all_entities.get("relationships", []) if r.get("type") == "CALLS"]))
```

Chame `resolve_pending_calls(entities)` dentro de `run_scan`, após todos os `scan_project` e antes do `ingest_to_neo4j`.

## 3. Sprint 2 — Métricas CK Suite Completo

### Por que isso importa?

As métricas Chidamber-Kemerer são o padrão acadêmico para medir qualidade OOP. O CAST usa exatamente elas. Todas derivam do AST que você já tem — sem precisar de IA.

### 3.1 Criar `backend/analyzers/ck_metrics.py`

```python
"""
ck_metrics.py
Implementa o suite Chidamber-Kemerer de métricas OOP.

Métricas implementadas:
- WMC  (Weighted Methods per Class)    — soma das complexidades dos métodos
- CBO  (Coupling Between Objects)      — quantas classes distintas esta usa
- RFC  (Response For a Class)          — WMC + métodos externos chamados
- LCOM (Lack of Cohesion in Methods)   — coesão interna da classe
- DIT  (Depth of Inheritance Tree)     — profundidade de herança
- NOC  (Number of Children)            — subclasses diretas

Referência: Chidamber & Kemerer, 1994 — "A Metrics Suite for Object Oriented Design"
"""

from dataclasses import dataclass, field
from typing import Optional


@dataclass
class CKMetrics:
    namespace_key: str
    class_name: str

    # Métricas básicas
    wmc: int = 0          # Soma das complexidades
    cbo: int = 0          # Acoplamento: qtd de classes externas usadas
    rfc: int = 0          # Métodos próprios + externos chamados
    lcom: float = 0.0     # Falta de coesão (0 = coeso, 1 = totalmente disperso)
    dit: int = 0          # Profundidade de herança
    noc: int = 0          # Número de subclasses
    loc: int = 0          # Linhas de código da classe

    # Métricas derivadas (calculadas em compute_derived)
    risk_score: float = 0.0       # Score final de risco (0-100)
    is_god_class: bool = False
    is_feature_envy: bool = False  # Usa mais métodos externos do que próprios
    maintainability_index: float = 100.0

    def compute_derived(self):
        """Calcula métricas derivadas após preencher as básicas."""
        # God Class: alta WMC + alto CBO
        self.is_god_class = self.wmc > 20 and self.cbo > 10

        # Feature Envy: RFC muito maior que WMC (usa mais de fora do que de dentro)
        if self.wmc > 0:
            self.is_feature_envy = (self.rfc - self.wmc) > self.wmc * 2

        # Maintainability Index (baseado no MI da Microsoft)
        # MI = max(0, (171 - 5.2*ln(HV) - 0.23*CC - 16.2*ln(LOC)) * 100/171)
        # Simplificado para nossas métricas:
        import math
        loc_safe = max(self.loc, 1)
        cc_proxy = self.wmc / max(len([1]), 1)  # Complexidade média por método
        self.maintainability_index = max(0, min(100,
            171 - 5.2 * math.log(max(self.rfc, 1))
                - 0.23 * cc_proxy
                - 16.2 * math.log(loc_safe)
        ) * 100 / 171)

        # Risk Score (0-100, maior = mais arriscado)
        # Fórmula própria calibrada para detectar pontos quentes
        self.risk_score = min(100, (
            self.wmc * 1.5 +          # Complexidade tem peso alto
            self.cbo * 2.0 +          # Acoplamento é crítico
            self.lcom * 20 +          # Falta de coesão
            max(0, self.dit - 3) * 5  # Herança profunda demais
        ) / 3)


class CKCalculator:
    """Calcula métricas CK a partir dos nós e arestas já parseados."""

    def calculate_for_class(
        self,
        class_ns_key: str,
        class_name: str,
        methods: list[dict],        # Lista de nós de método desta classe
        relationships: list[dict],  # Todas as arestas do sistema
        all_nodes: list[dict],      # Todos os nós para resolver DIT
    ) -> CKMetrics:

        metrics = CKMetrics(
            namespace_key=class_ns_key,
            class_name=class_name
        )

        method_keys = {m["namespace_key"] for m in methods}

        # ─── WMC: soma das complexidades ───
        metrics.wmc = sum(m.get("complexity", 1) for m in methods)
        metrics.loc = sum(m.get("loc", 0) for m in methods)

        # ─── CBO: classes externas usadas ───
        external_classes = set()
        for rel in relationships:
            if rel.get("from") in method_keys:
                target_key = rel.get("to", "")
                # Extrai o nome da classe do namespace_key: project:file:Classe.metodo
                parts = target_key.split(":")
                if len(parts) >= 3:
                    class_part = parts[2].split(".")[0]
                    if class_part != class_name:
                        external_classes.add(class_part)
        metrics.cbo = len(external_classes)

        # ─── RFC: WMC + chamadas externas únicas ───
        external_methods = set()
        for rel in relationships:
            if rel.get("from") in method_keys and rel.get("type") == "CALLS":
                external_methods.add(rel.get("to", ""))
        metrics.rfc = metrics.wmc + len(external_methods)

        # ─── LCOM: falta de coesão ───
        # Heurística simplificada: se CBO >> número de métodos, coesão é baixa
        if len(methods) > 0:
            max_possible_coupling = len(methods) * 5
            metrics.lcom = min(1.0, metrics.cbo / max(max_possible_coupling, 1))

        # ─── DIT: profundidade de herança ───
        # Detectamos isso durante o parse quando salvamos "extends"
        # Por ora, usamos 1 como default (maioria não tem herança profunda)
        metrics.dit = 1

        metrics.compute_derived()
        return metrics


def calculate_all_metrics(
    nodes: list[dict],
    relationships: list[dict]
) -> dict[str, CKMetrics]:
    """
    Calcula métricas CK para todas as classes Java/TS no grafo.
    Retorna dict: namespace_key → CKMetrics
    """
    calculator = CKCalculator()
    results = {}

    # Agrupa métodos por classe
    class_nodes = [n for n in nodes if "Java_Class" in n.get("labels", []) or "TS_Component" in n.get("labels", [])]
    method_nodes = [n for n in nodes if "Java_Method" in n.get("labels", []) or "TS_Function" in n.get("labels", [])]

    # Índice: parent_class → lista de métodos
    methods_by_class: dict[str, list[dict]] = {}
    for method in method_nodes:
        parent = method.get("parent_class", "")
        if parent:
            methods_by_class.setdefault(parent, []).append(method)

    for cls in class_nodes:
        class_name = cls.get("name", "")
        class_key = cls.get("namespace_key", "")
        methods = methods_by_class.get(class_name, [])

        if not methods:
            continue

        metrics = calculator.calculate_for_class(
            class_key, class_name, methods, relationships, nodes
        )
        results[class_key] = metrics

    return results
```

### 3.3 Novo endpoint para métricas CK

Adicione no `main.py`:

```python
@app.get("/api/metrics/ck")
async def get_ck_metrics(project: str = None, min_risk: float = 0):
    """Retorna métricas CK de todas as classes, ordenadas por risk_score."""
    nodes = memory_nodes if not neo4j_service.is_connected else []

    if neo4j_service.is_connected:
        try:
            where = "WHERE n.risk_score >= $min_risk"
            params = {"min_risk": min_risk}
            if project:
                where += " AND n.project = $project"
                params["project"] = project

            result = neo4j_service.graph.run(f"""
                MATCH (n:Entity)
                {where}
                AND n.wmc IS NOT NULL
                RETURN n.namespace_key AS key, n.name AS name,
                       n.layer AS layer, n.project AS project,
                       n.wmc AS wmc, n.cbo AS cbo, n.rfc AS rfc,
                       n.lcom AS lcom, n.risk_score AS risk_score,
                       n.maintainability_index AS mi,
                       n.is_god_class AS god_class,
                       n.loc AS loc
                ORDER BY n.risk_score DESC
                LIMIT 200
            """, **params).data()
            return {"metrics": result, "total": len(result)}
        except Exception as e:
            logger.warning("Fallback para memória nas métricas CK: %s", e)

    # Fallback memória
    filtered = [
        n for n in memory_nodes
        if n.get("risk_score", 0) >= min_risk
        and n.get("wmc") is not None
        and (not project or n.get("project") == project)
    ]
    filtered.sort(key=lambda x: x.get("risk_score", 0), reverse=True)
    return {"metrics": filtered[:200], "total": len(filtered)}
```

```python
"""
git_analyzer.py
Analisa histórico Git para detectar hotspots de código.

Baseado no trabalho de Adam Tornhill ("Your Code as a Crime Scene", 2015).
A hipótese central: arquivo com ALTA COMPLEXIDADE + MUITAS MUDANÇAS = maior risco real.
Um arquivo perfeito que nunca muda não é problema. Um arquivo bagunçado que muda todo dia é.
"""

import subprocess
import os
import re
from datetime import datetime, timedelta
from pathlib import Path
from dataclasses import dataclass, field


@dataclass
class FileChurnData:
    """Dados de churn (mudanças) de um arquivo."""
    file_path: str
    change_count: int = 0           # Quantas vezes foi modificado
    authors: set = field(default_factory=set)  # Autores que modificaram
    last_modified: str = ""         # Data da última modificação
    age_days: int = 0               # Idade do arquivo em dias
    lines_added: int = 0            # Total de linhas adicionadas no período
    lines_removed: int = 0          # Total de linhas removidas no período


@dataclass
class HotspotScore:
    """Score de hotspot combinando complexidade e churn."""
    file_path: str
    complexity: float = 0.0         # Complexidade média dos métodos no arquivo
    churn: int = 0                  # Número de mudanças no período
    authors: int = 0                # Número de autores distintos
    hotspot_score: float = 0.0      # Score final (maior = mais perigoso)
    category: str = "normal"        # normal / watch / danger / critical


class GitAnalyzer:
    """Analisa o histórico git de um repositório."""

    def __init__(self, repo_path: str, days_back: int = 90):
        self.repo_path = repo_path
        self.days_back = days_back
        self._is_git_repo = self._check_git_repo()

    def _check_git_repo(self) -> bool:
        """Verifica se o diretório é um repositório git."""
        try:
            result = subprocess.run(
                ["git", "rev-parse", "--git-dir"],
                cwd=self.repo_path,
                capture_output=True, text=True, timeout=5
            )
            return result.returncode == 0
        except Exception:
            return False

    def get_file_churn(self) -> dict[str, FileChurnData]:
        """
        Retorna dados de churn por arquivo nos últimos N dias.
        Usa git log com --name-only para eficiência.
        """
        if not self._is_git_repo:
            return {}

        churn_data: dict[str, FileChurnData] = {}

        try:
            # Comando 1: arquivos modificados e quando
            result = subprocess.run(
                ["git", "log",
                 f"--since={self.days_back} days ago",
                 "--name-only",
                 "--format=COMMIT|%ae|%ai",  # email do autor + data ISO
                 "--diff-filter=M"],           # apenas modificações (não criações)
                cwd=self.repo_path,
                capture_output=True, text=True, timeout=30
            )

            current_author = ""
            current_date = ""

            for line in result.stdout.splitlines():
                if line.startswith("COMMIT|"):
                    parts = line.split("|")
                    current_author = parts[1] if len(parts) > 1 else "unknown"
                    current_date = parts[2][:10] if len(parts) > 2 else ""
                elif line.strip() and not line.startswith("COMMIT"):
                    file_path = line.strip()
                    # Normaliza separadores
                    file_path = file_path.replace("\\", "/")

                    if file_path not in churn_data:
                        churn_data[file_path] = FileChurnData(file_path=file_path)

                    churn_data[file_path].change_count += 1
                    churn_data[file_path].authors.add(current_author)
                    if not churn_data[file_path].last_modified:
                        churn_data[file_path].last_modified = current_date

            # Comando 2: idade dos arquivos (primeiro commit de cada um)
            result2 = subprocess.run(
                ["git", "log", "--diff-filter=A",
                 "--format=%ai %n--FILES--",
                 "--name-only"],
                cwd=self.repo_path,
                capture_output=True, text=True, timeout=30
            )
            # (simplificado — parseamos só o que precisamos)

        except subprocess.TimeoutExpired:
            pass
        except Exception as e:
            pass  # Retorna o que temos até agora

        return churn_data

    def calculate_hotspots(
        self,
        memory_nodes: list[dict],
        churn_data: dict[str, FileChurnData]
    ) -> list[HotspotScore]:
        """
        Combina complexidade do AST com churn do Git para gerar hotspot scores.

        Fórmula: hotspot = complexity_score * log(1 + churn_count) * author_factor
        """
        hotspots = []

        # Agrupa nós por arquivo
        file_metrics: dict[str, dict] = {}
        for node in memory_nodes:
            file_path = node.get("file", "")
            if not file_path:
                continue
            file_path = file_path.replace("\\", "/")

            if file_path not in file_metrics:
                file_metrics[file_path] = {
                    "complexities": [],
                    "locs": [],
                    "project": node.get("project", ""),
                }
            if node.get("complexity"):
                file_metrics[file_path]["complexities"].append(node["complexity"])
            if node.get("loc"):
                file_metrics[file_path]["locs"].append(node["loc"])

        import math

        for file_path, metrics in file_metrics.items():
            churn = churn_data.get(file_path)

            complexities = metrics["complexities"]
            avg_complexity = sum(complexities) / len(complexities) if complexities else 1.0
            max_complexity = max(complexities) if complexities else 1.0

            churn_count = churn.change_count if churn else 0
            author_count = len(churn.authors) if churn else 1

            # Score principal: baseado em Tornhill
            # Normalizado: complexidade 0-1 * churn logarítmico
            complexity_score = min(1.0, (avg_complexity + max_complexity) / 40)
            churn_score = math.log(1 + churn_count) / math.log(1 + 100)  # normaliza para 100 mudanças
            author_factor = 1 + (author_count - 1) * 0.1  # mais autores = mais risco de inconsistência

            hotspot_raw = complexity_score * churn_score * author_factor * 100

            # Categoria
            if hotspot_raw >= 70:
                category = "critical"
            elif hotspot_raw >= 40:
                category = "danger"
            elif hotspot_raw >= 20:
                category = "watch"
            else:
                category = "normal"

            hotspots.append(HotspotScore(
                file_path=file_path,
                complexity=round(avg_complexity, 2),
                churn=churn_count,
                authors=author_count,
                hotspot_score=round(hotspot_raw, 1),
                category=category,
            ))

        # Ordena por score decrescente
        hotspots.sort(key=lambda h: h.hotspot_score, reverse=True)
        return hotspots
```

### 5.1 Criar `backend/analyzers/rag_engine.py`

```python
"""
rag_engine.py
RAG (Retrieval-Augmented Generation) sobre o grafo de dependências.

Em vez de mandar 15 nós aleatórios para o LLM, indexamos todos os nós
como embeddings e recuperamos apenas os semanticamente relevantes
para cada pergunta. Resultado: respostas muito mais precisas.

Modelo: nomic-embed-text (300MB, ultraleve, roda paralelo com o chat model)
"""

import json
import asyncio
import logging
import httpx
import numpy as np
from typing import Optional

logger = logging.getLogger("insightgraph.rag")

OLLAMA_URL = "http://localhost:11434"
EMBED_MODEL = "nomic-embed-text"


class RAGEngine:
    """Motor de busca semântica sobre os nós do grafo."""

    def __init__(self):
        self.embeddings: dict[str, list[float]] = {}  # namespace_key → embedding
        self.node_texts: dict[str, str] = {}          # namespace_key → texto descritivo
        self._indexed = False

    def _node_to_text(self, node: dict) -> str:
        """
        Converte um nó do grafo em texto descritivo para embedding.
        Quanto mais contexto, melhor a busca semântica.
        """
        parts = []
        name = node.get("name", "")
        labels = [l for l in node.get("labels", []) if l != "Entity"]
        layer = node.get("layer", "")
        project = node.get("project", "")
        file = node.get("file", "")
        decorators = node.get("decorators", "")
        parent = node.get("parent_class", "")

        if labels:
            parts.append(f"Type: {labels[0]}")
        parts.append(f"Name: {name}")
        if parent:
            parts.append(f"In class: {parent}")
        if layer:
            parts.append(f"Layer: {layer}")
        if project:
            parts.append(f"Project: {project}")
        if file:
            parts.append(f"File: {file}")
        if decorators:
            parts.append(f"Annotations: {decorators}")

        # Adiciona complexidade se disponível
        if node.get("complexity", 0) > 5:
            parts.append(f"High complexity: {node['complexity']}")
        if node.get("risk_score", 0) > 30:
            parts.append(f"Risk score: {node['risk_score']}")

        return " | ".join(parts)

    async def _embed_text(self, text: str) -> Optional[list[float]]:
        """Gera embedding via Ollama nomic-embed-text."""
        try:
            async with httpx.AsyncClient(timeout=10.0) as client:
                resp = await client.post(
                    f"{OLLAMA_URL}/api/embeddings",
                    json={"model": EMBED_MODEL, "prompt": text}
                )
                resp.raise_for_status()
                return resp.json().get("embedding")
        except Exception as e:
            logger.warning("Falha ao gerar embedding: %s", e)
            return None

    async def build_index(self, nodes: list[dict], batch_size: int = 20) -> None:
        """
        Indexa todos os nós do grafo em embeddings.
        Roda em batches para não sobrecarregar o Ollama.
        """
        logger.info("Construindo índice RAG para %d nós...", len(nodes))
        self.embeddings = {}
        self.node_texts = {}

        # Filtra nós relevantes (pula externos e muito pequenos)
        relevant_nodes = [
            n for n in nodes
            if n.get("namespace_key") and n.get("name")
            and "External" not in n.get("labels", [])
        ]

        for i in range(0, len(relevant_nodes), batch_size):
            batch = relevant_nodes[i:i + batch_size]
            tasks = []
            for node in batch:
                key = node["namespace_key"]
                text = self._node_to_text(node)
                self.node_texts[key] = text
                tasks.append(self._embed_text(text))

            results = await asyncio.gather(*tasks, return_exceptions=True)

            for node, result in zip(batch, results):
                if isinstance(result, list):
                    self.embeddings[node["namespace_key"]] = result

            if i % 100 == 0:
                logger.info("RAG: indexados %d/%d nós", i, len(relevant_nodes))

        self._indexed = True
        logger.info("Índice RAG concluído: %d embeddings", len(self.embeddings))

    def cosine_similarity(self, a: list[float], b: list[float]) -> float:
        """Similaridade coseno entre dois vetores."""
        va = np.array(a)
        vb = np.array(b)
        norm_a = np.linalg.norm(va)
        norm_b = np.linalg.norm(vb)
        if norm_a == 0 or norm_b == 0:
            return 0.0
        return float(np.dot(va, vb) / (norm_a * norm_b))

    async def search(self, query: str, top_k: int = 15) -> list[dict]:
        """
        Busca os nós mais semanticamente relevantes para a query.
        Retorna lista de (namespace_key, texto, score).
        """
        if not self._indexed or not self.embeddings:
            return []

        query_embedding = await self._embed_text(query)
        if not query_embedding:
            return []

        scores = []
        for key, emb in self.embeddings.items():
            score = self.cosine_similarity(query_embedding, emb)
            scores.append((key, score))

        scores.sort(key=lambda x: x[1], reverse=True)
        top = scores[:top_k]

        return [
            {"key": key, "text": self.node_texts.get(key, ""), "score": round(score, 4)}
            for key, score in top
        ]


# Instância global do RAG engine
rag_engine = RAGEngine()
```

### Endpoint de busca semântica (bônus):

```python
@app.get("/api/graph/search")
async def semantic_search(q: str, top_k: int = 20):
    """
    Busca semântica no grafo: retorna os nós mais relevantes para o termo.
    Usa o índice RAG com embeddings nomic-embed-text.
    """
    if not rag_engine._indexed:
        return {"results": [], "message": "Índice RAG ainda não construído. Execute um scan primeiro."}

    results = await rag_engine.search(q, top_k=top_k)

    # Enriquece com dados dos nós
    enriched = []
    node_map = {n["namespace_key"]: n for n in memory_nodes}
    for r in results:
        node = node_map.get(r["key"], {})
        enriched.append({
            **r,
            "name": node.get("name", ""),
            "layer": node.get("layer", ""),
            "project": node.get("project", ""),
            "file": node.get("file", ""),
            "labels": node.get("labels", []),
        })

    return {"results": enriched, "query": q}
```
